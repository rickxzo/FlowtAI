async function send() { const input = document.getElementById("msg"); const userText = input.value.trim(); if (!userText) return; addUserMessage(userText); input.value = ""; const bubble = createBotMessageContainer(); let isFinal = false; let finalTextNode = null; function appendText(node, text) { const trimmed = text.trim(); if (!trimmed) return; // Add space only if needed if ( node.textContent.length > 0 && !node.textContent.endsWith(" ") ) { node.textContent += " "; } node.textContent += trimmed; } try { const res = await fetch( `https://flowtai-1.onrender.com/respond?agent_id=73ed20a1-091a-4b5a-a676-ad86377d8bad&input=${encodeURIComponent(userText)}&convo=67` ); const reader = res.body.getReader(); const decoder = new TextDecoder(); while (true) { const { done, value } = await reader.read(); if (done) break; const chunk = decoder.decode(value, { stream: true }); const lines = chunk.split("\n").filter(line => line.trim() !== ""); for (let line of lines) { if (line.startsWith("data:")) { isFinal = true; const clean = line.replace(/^data:\s*/, ""); if (!finalTextNode) { finalTextNode = document.createElement("div"); finalTextNode.className = "text-zinc-800"; bubble.appendChild(finalTextNode); } appendText(finalTextNode, clean); } else { if (!isFinal) { // intermediate thoughts const step = document.createElement("div"); step.className = "text-zinc-400 text-xs"; step.textContent = line; bubble.appendChild(step); } else { // final streaming continuation if (!finalTextNode) { finalTextNode = document.createElement("div"); finalTextNode.className = "text-zinc-800"; bubble.appendChild(finalTextNode); } appendText(finalTextNode, line); } } } const chat = document.getElementById("chat"); chat.scrollTop = chat.scrollHeight; } } catch (err) { bubble.innerHTML = `⚠️ Error streaming response`; } }
